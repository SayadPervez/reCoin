string Return_Wholefile()
{
    ifstream fin;
    fin.open("C:\\Users\\admin\\OneDrive\\Documents\\first.txt");
    string sLine;
    string tLine;
    if(!fin)
    {
        cout<< "File Not Found";
        exit(1);
    }
    else
    while(!fin.eof())
    {
      getline(fin, sLine);
      tLine.append(sLine);
      tLine.append(" ");
    }
    fin.close();
    return tLine;
}
